#include <stdio.h>

int main() {
    char nome1[50], nome2[50];
    int idade1, idade2;

    // Solicita os dados do usuário
    printf("Digite o nome da primeira pessoa: ");
    fgets(nome1, sizeof(nome1), stdin);
    // Remove o newline no final do nome, se presente
    nome1[strcspn(nome1, "\n")] = '\0';

    printf("Digite a idade da primeira pessoa: ");
    scanf("%d", &idade1);

    // Limpa o buffer do teclado
    getchar();

    printf("Digite o nome da segunda pessoa: ");
    fgets(nome2, sizeof(nome2), stdin);
    // Remove o newline no final do nome, se presente
    nome2[strcspn(nome2, "\n")] = '\0';

    printf("Digite a idade da segunda pessoa: ");
    scanf("%d", &idade2);

    // Determina e exibe a pessoa mais nova
    if (idade1 < idade2) {
        printf("A pessoa mais nova é %s com %d anos.\n", nome1, idade1);
    } else if (idade2 < idade1) {
        printf("A pessoa mais nova é %s com %d anos.\n", nome2, idade2);
    } else {
        printf("Ambas as pessoas têm a mesma idade de %d anos.\n", idade1);
    }

    return 0;
}
